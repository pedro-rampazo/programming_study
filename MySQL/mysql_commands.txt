MySQL installation:

$ sudo apt update
$ sudo apt install mysql-server

MySQL configuration:

$ sudo mysql_secure_installation

MySQL testing:

$ systemctl status mysql → show MySQL status
$ sudo systemctl start mysql → start MySQL
$ sudo systemctl stop mysql → stop MySQL
$ mysql -u root -p

------------------------------------------
DATABASE → TABLE → ROWS

#VIEWING DATABASES#

mysql> show databases;                                      {view database}
mysql> use <database>;                                      {change my working database}
mysql> show tables;                                         {view tables}
mysql> show columns from `<table>`;                         {show table columns}
mysql> select * from `<table>`;                             {show table rows}
mysql> select * from `<table>` order by `<column>` desc;    {sort in descending order}
mysql> select * from `<table>` order by by `<column>` asc;  (sort in ascending order)

#MANIPULATING DATABASES#

mysql> create database <database>; {create a new database}

mysql> create table `<table>` (`id` INT NOT NULL AUTO_INCREMENT, `<column>` <DATA TYPE> <NULL>,
        `<column>` <DATA TYPE> <NULL>, `<column>` <DATA TYPE> <NULL>, PRIMARY KEY (`id`))
        ENGINE=InnoDB; {create table with columns}

mysql> insert into `<table>` VALUES (null, "<content>", "<content>", <content>); {Insert a row in the table}

mysql> update `<table>` set <column> = '<value>' where id = 1; {update values in the table}

mysql> delete from `<table>` where <column> = '<value>' and <column> = <value>; {delete values in the table}
